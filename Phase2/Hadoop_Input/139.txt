      <script type="text/javascript">
  history.navigationMode = 'compatible';

  var t, st;
  var lastidx;
  var ishover = false;
  var isscrolling = false;
  var tcar;
  var nbritems = 0;
  var idlist  = new Array();
  var vsl = 1;
  var sauto = 8;
  var sshow = 3;

function mycarousel_itemFirstInCallback(carousel, item, idx, state) {
  var idm = ((idx-1) % nbritems) + 1;
  if (idm <= 0) idm = nbritems + idm;
  $('#slider').slider('option', 'value', idm);
  lastidx = idx;
  $("#idxdata").html(lastidx + " (" + idm + ") /" + nbritems);
  return true;
};


function mycarousel_resetAuto() {
  if (ishover) {
    t = setTimeout("mycarousel_resetAuto();", 6000);
  } else {
    tcar.startAuto(sauto);
  }
  return true;
};

function mycarousel_setSlider() {
  clearTimeout(st);
  if (isscrolling) {
	$("#sldata").html("scrolling " + vsl);
    st = setTimeout("mycarousel_setSlider();", 250);
  } else {
	$("#sldata").html("set " + vsl);
    tcar.scroll(parseInt(vsl),false);
  }
  return true;
};

function mycarousel_initCallback(carousel) {
  tcar = carousel;
  if (nbritems > sshow ) {

	$('#slider').bind('slide', function(event,ui) {
		clearTimeout(st);
		clearTimeout(t);
		carousel.startAuto(0);
		vsl = ui.value;
		if (isscrolling) {
		  $("#sldata").html("scrolling " + vsl);
          st = setTimeout("mycarousel_setSlider();", 250);
		} else {
		  mycarousel_setSlider();
		}

		t = setTimeout("mycarousel_resetAuto();", 6000);
    });


    // Disable autoscrolling if the user clicks the prev or next button.
    carousel.buttonNext.bind('click', function() {
		clearTimeout(t);
		t = setTimeout("mycarousel_resetAuto();", 6000);
        carousel.startAuto(0);
    });

    carousel.buttonPrev.bind('click', function() {
		clearTimeout(t);
		t = setTimeout("mycarousel_resetAuto();", 6000);
        carousel.startAuto(0);
    });

    // Pause autoscrolling if the user moves with the cursor over the clip.
    //carousel.clip.hover(function() {
    $('#mycarousel').hover(function() {
        clearTimeout(t);
        ishover = true;
        //carousel.stopAuto();
		carousel.startAuto(0);
		$("#hdata").html("is hover");
    }, function() {
		clearTimeout(t);
		$("#hdata").html("is not hover");
		ishover = false;
        //carousel.startAuto(3);
		t = setTimeout("mycarousel_resetAuto();", 6000);
    });
  } else {
    carousel.startAuto(0);
    $("#slider").hide();
    $("#news_carousel").find(".jcarousel-next").hide();
    $("#news_carousel").find(".jcarousel-prev").hide();
  }
  return true;

};


function mycarousel_itemVisibleInCallback(carousel, item, i, state, evt) {
  // The index() method calculates the index from a
