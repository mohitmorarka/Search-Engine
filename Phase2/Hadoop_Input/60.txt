      <script type="text/javascript">
  history.navigationMode = 'compatible';

  var t, st;
  var lastidx;
  var ishover = false;
  var isscrolling = false;
  var tcar;
  var nbritems = 0;
  var idlist  = new Array();
  var vsl = 1;
  var sauto = 8;
  var sshow = 3;

function mycarousel_itemFirstInCallback(carousel, item, idx, state) {
  var idm = ((idx-1) % nbritems) + 1;
  if (idm <= 0) idm = nbritems + idm;
  $('#slider').slider('option', 'value', idm);
  lastidx = idx;
  $("#idxdata").html(lastidx + " (" + idm + ") /" + nbritems);
  return true;
};


function mycarousel_resetAuto() {
  if (ishover) {
    t = setTimeout("mycarousel_resetAuto();", 6000);
  } else {
    tcar.startAuto(sauto);
  }
  return true;
};

function mycarousel_setSlider() {
  clearTimeout(st);
  if (isscrolling) {
	$("#sldata").html("scrolling " + vsl);
    st = setTimeout("mycarousel_setSlider();", 250);
  } else {
	$("#sldata").html("set " + vsl);
    tcar.scroll(parseInt(vsl),false);
  }
  return true;
};

function mycarousel_initCallback(carousel) {
  tcar = carousel;
  if (nbritems > sshow ) {

	$('#slider').bind('slide', function(event,ui) {
		clearTimeout(st);
		clearTimeout(t);
		carousel.startAuto(0);
		vsl = ui.value;
		if (isscrolling) {
		  $("#sldata").html("scrolling " + vsl);
          st = setTimeout("mycarousel_setSlider();", 250);
		} else {
		  mycarousel_setSlider();
		}

		t = setTimeout("mycarousel_resetAuto();", 6000);
    });


    // Disable autoscrolling if the user clicks the prev or next button.
    carousel.buttonNext.bind('click', function() {
		clearTimeout(t);
		t = setTimeout("mycarousel_resetAuto();", 6000);
        carousel.startAuto(0);
    });

    carousel.buttonPrev.bind('click', function() {
		clearTimeout(t);
		t = setTimeout("mycarousel_resetAuto();", 6000);
        carousel.startAuto(0);
    });

    // Pause autoscrolling if the user moves with the cursor over the clip.
    //carousel.clip.hover(function() {
    $('#mycarousel').hover(function() {
        clearTimeout(t);
        ishover = true;
        //carousel.stopAuto();
		carousel.startAuto(0);
		$("#hdata").html("is hover");
    }, function() {
		clearTimeout(t);
		$("#hdata").html("is not hover");
		ishover = false;
        //carousel.startAuto(3);
		t = setTimeout("mycarousel_resetAuto();", 6000);
    });
  } else {
    carousel.startAuto(0);
    $("#slider").hide();
    $("#news_carousel").find(".jcarousel-next").hide();
    $("#news_carousel").find(".jcarousel-prev").hide();
  }
  return true;

};


function mycarousel_itemVisibleInCallback(carousel, item, i, state, evt) {
  // The index() method calculates the index from a
  // given index who is out of the actual item range.
  $("#iodata").html("in " + state);
  if (state != "init") isscrolling = true;
  var idx = carousel.index(i, idlist.length) - 1;
  var izz = idlist[idx];
  carousel.add(i, mycarousel_getItemHTML(izz));
  return true;
};

function mycarousel_itemVisibleOutCallback(carousel, item, i, state, evt) {
    carousel.remove(i);
	isscrolling = false;
	$("#iodata").html("out " + state);
  return true;
};

/**
 * Item html creation helper.
 */
function mycarousel_getItemHTML(item) {
  return '<div "id="newsactive' + item + '" >' + $("#newsid" + item).html() + '</div>';
};

     <h2 class="intro"> Be Safe. Be Well. Be a Highlander. </h2> 
     <h3 class="verd intro"> Our students are strong in body, mind and spirit — thanks<br /> to programs that support your total-body wellness. </h3> 
     <div class="grid_10 prefix_1 suffix_1 alpha omega"> 
      <div class="clearfix" style="margin-bottom: 25px;"> 
       <div class="grid_5 alpha"> 
        <div class="doubleRoundedCorners borderGrey"> 
         <div class="clearfix gradientGrey" style="padding: 5px 0 5px 10px; height: 165px;"> 
          <img alt="Arm curl" style="margin-right: -5px;" class="imgFloatRight " width="102" height="163" src="/Images/armCurl.png" /> 
          <h3 style="margin: 10px 0 8px;"> Pumped!</h3> 
          <p style="font-size: 1em;"> Learn about the new Student Recreation Center with its huge swimming pool, indoor track and triple the weights. </p> 
          <p style="margin-top: 8px;"> <a class="rightArrowCTA" href="http://www.recreation.ucr.edu/SRC_expansion/Pages/default.aspx"> Opening in 2014</a></p> 
         </div> 
        </div> 
       </div> 
       <div class="grid_5 omega"> 
        <div class="doubleRoundedCorners borderGrey"> 
         <div class="clearfix gradientGrey" style="padding: 5px 0 5px 10px; height: 165px;"> 
          <img alt="" class="imgFloatRight " width="112" height="176" src="/Images/stretch.png" style="position: relative; bottom: 7px;" /> 
          <h3 style="margin: 10px 0 8px;"> Stretch!</h3> 
          <p style="font-size: 1em;"> Stretch yourself with our invigorating recreational student organizations. </p> 
          <p style="margin-top: 8px;"> <a class="rightArrowCTA" href="http://highlanderlink.ucr.edu/organizations?SearchValue=Recreational&amp;SearchType=Category&amp;CurrentPage=1&amp;SelectedCategoryId=4173">See student orgs that support your wellness.</a></p> 
         </div> 
        </div> 
       </div> 
      </div> 
      <div class="clearfix iconSections version2 alternatingBoxes"> 
       <div> 
        <h3> <span> <img src="/Images/icons/safeIcon.png" alt="" /> </span>Be Safe </h3> 
        <p> Live in <a href="http://housing.ucr.edu/liveUCR/campus-safety.aspx">secure residence halls</a> near our <a href="http://police.ucr.edu/">on-campus police department</a>. Get texts from a <a href="http://cnc.ucr.edu/ens/">sophisticated emergency notification system</a> and cross campus safely at night with our <a href="http://wrc.ucr.edu/escortservices/Pages/default.aspx">award-winning Campus Safety Escorts</a><a href="http://escortservice.ucr.edu/"></a>. See <a href="http://www.ucr.edu/parents/safety.html"> more safety information for parents</a>.</p> 
       </div> 
       <div> 
        <h3> <span> <img src="/Images/icons/healthyIcon.png" alt="" /></span> Be Healthy</h3> 
        <p> Our <a href="http://www.campushealth.ucr.edu/Pages/default.aspx">Campus Health Center</a> offers everything from walk-up and urgent care, to dental and optical. If you have the <a href="http://campushealth.ucr.edu/insurance/Pages/Summary.aspx">Undergraduate Student Health Insurance Plan</a><a href="https://wfis.wellsfargo.com/ProductServices/A%20to%20Z/StudentInsurance/ucship/"></a>, doctor visits are free. </p> 
       </div> 
       <div> 
        <h3> <span> <img src="/Images/icons/supportIcon.png" alt="" /> </span>Be Supported</h3> 
        <p> Access free and affordable <a href="http://counseling.ucr.edu/counseling-services/Pages/default.aspx"> confidential counseling</a> and <a href="http://counseling.ucr.edu/emergency-assistance/Pages/default.aspx"> crisis support</a> at our <a href="http://counseling.ucr.edu/Pages/default.aspx">Counseling Center</a>. Stressed out? Get <a href="http://counseling.ucr.edu/counseling-services/Pages/Biofeedback.aspx"> biofeedback</a> or call <a href="http://counseling.ucr.edu/counseling-services/stressbusters/Pages/default.aspx"> the StressBusters</a>. You can reach a counselor <a href="http://counseling.ucr.edu/emergency-assistance/Pages/default.aspx"> by phone 24/7</a>.</p> 
       </div> 
       <div> 
        <h3> <span> <img src="/Images/icons/nourishedIcon.png" alt="" /> </span>Be Nourished</h3> 
        <p> Find fresh delights at our <a href="http://wellness.ucr.edu/healthy_eating_on_campus.html">on-campus Farmers' Market</a><a href="http://www.insideucr.ucr.edu/cgi-bin/display.cgi?id=1500"></a> or peruse <a href="http://wellness.ucr.edu/healthy_vending.html"> vending machines</a> with at least 25% healthy snacks. Enjoy organic options in <a href="http://dining.ucr.edu/MarketsConvenienceStores/Pages/MarketsConvenienceStores.aspx"> campus stores</a>, choose from “Lighten Up” menus and dig into meals prepared with local produce. </p> 
       </div> 
       <div> 
        <h3> <span> <img src="/Images/icons/challengedIcon.png" alt="" /> </span>Be Challenged</h3> 
        <p> Explore Yosemite, kayak Mission Bay, snorkel Catalina and surf San Clemente with <a href="http://www.recreation.ucr.edu/outdoor_excursions/Pages/OutdoorExcursionsHome.aspx"> UCR Outdoor Excursions.</a> Hike nearby Mt. Rubidoux (minutes away) or head to the exclusive <a href="http://www.recreation.ucr.edu/Big_Bear/Pages/default.aspx"> UCR Lake House at Big Bear</a> for some fun in the snow. </p> 
       </div> 
       <div> 
        <h3> <span> <img src="/Images/icons/fitIcon.png" alt="" /> </span>Be Fit</h3> 
        <p> Work out in our 80,000-square-foot <a href="http://www.recreation.ucr.edu/Pages/default.aspx"> Student Recreation Center</a> (SRC) offering personal training and recreation programs in everything from intramural flag football to bagpipes and dance. Skate on the roller hockey rink, bump and set in sand volleyball courts and scale the rock wall. UCR has <a href="http://www.recreation.ucr.edu/competitiveSports/Pages/SportsClubs.aspx"> several sports clubs</a> including rugby, soccer and Tae Kwon Do. SRC membership included in your student fees. </p> 
       </div> 
       <div> 
        <h3> <span> <img src="/Images/icons/wellIcon.png" alt="" /> </span>Be Well</h3> 
        <p> The Well is a unique UCR resource committed to your <a href="http://well.ucr.edu/healthandwellness/Pages/YourHealthandWellness.aspx"> physical, social, emotional, cultural and spiritual wellness.&nbsp;</a>Take a meditation class, win a prize with your pedometer, or snooze through the <a href="http://ucrtoday.ucr.edu/3723"> “No Stress Sleep Fest.”</a> Connect to <a href="http://well.ucr.edu/diversityresources/Pages/DiversityPrograms.aspx"> diversity programs</a>, <a href="http://well.ucr.edu/communityservice/Pages/communityservice.aspx"> community service</a> and <a href="http://well.ucr.edu/Peers/Pages/Peers.aspx">peer education</a>. Feel good. </p> 
       </div> 
      </div>

      <h3>Welcome to the Department of Nematology at the University of California, Riverside.</h3> 
      <p>As one of very few Nematology Departments, we have drawn upon expertise from seven countries to build a diverse faculty of nine, addressing nematology research, teaching, and outreach in relation to agriculture, biology, genetics, systematics, ecology, evolution, and parasitology. Our continuing ties with the Agricultural Experiment Station are reflected by our particular strengths in developing environmentally sound nematode management practices including resistance, cultural practices and biological and chemical control. Our faculty interact across departments and programs to provide an abundance of opportunities for undergraduate, graduate and postdoctoral training as well as a welcoming stimulating environment for visiting scientists. Whether you're a web surfer, colleague, or prospective graduate student, we are eager to share, through these pages, a sense of our enthusiasm for the work of our department and for the exciting field of Nematology.</p>
