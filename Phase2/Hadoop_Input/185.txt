       <p>RIVERSIDE, Calif. — The University of California, Riverside has once-again been recognized as a “Best Value School” and was lauded for its diversity in the <a href="http://www.usnews.com/colleges" onclick="javascript:_gaq.push(['_trackEvent','outbound-article','http://www.usnews.com/colleges']);">U.S. News and World Report’s 2014 Best Colleges Guide</a>, released on Tuesday, Sept. 10.</p> 
       <p>“We know that rankings like the U.S. News and World Report and Washington Monthly surveys are widely followed by potential students as well as current students, alumni, faculty and staff,” said UC Riverside Executive Vice Chancellor and Provost Dallas Rabenstein. “As a result, we are pleased to once again be ranked as one of the nation’s top research universities.”</p> 
       <p>Overall, UCR is ranked 112th overall among national universities, and 55th among public universities. The numbers mark a slight decline from the 101st and 47th rankings in the 2013 survey.</p> 
       <p>The rankings, which are based on categories developed by the Carnegie Foundation for the Advancement of Teaching, placed UCR as 35th among the 50 schools on the best value list, which considers academic quality and affordability. UCR was 24th in the category in the 2013 survey.</p> 
       <p>The university once again appeared on the list of the most diverse schools in the nation, placing in a tie for 12th overall. The score of 0.70 was identical to the 2013 survey, where UCR finished in a tie for 8th place.</p> 
       <p>Rabenstein said he wasn’t concerned with UCR’s slightly lower placement when compared to past years, citing the example of the ethnic diversity category, in which just seven-hundredths of a point separated the top 17 schools in the category.</p> 
       <p>“Because the differences between campuses can be so incremental, it’s not unusual to see volatility in the rankings from one year to the next,” he said.</p> 
       <p>The annual&nbsp;rankings&nbsp;by U.S. News and World Report, first issued in 1983, are frequently used by families as they decide where to target applications. The rankings combine reputation-based rankings with the hard-data outcomes related to graduation rates, tuition costs and level of donor support.</p> 
       <p>UC Riverside has fared well in the annual surveys, including:</p> 
       <ul> 
        <li>In August 2013, Washington Monthly magazine ranked the University of California, Riverside second among national universities in their <a href="http://www.washingtonmonthly.com/college_guide/toc_2013.php" onclick="javascript:_gaq.push(['_trackEvent','outbound-article','http://www.washingtonmonthly.com/college_guide/toc_2013.php']);">ninth-annual College Ranking Survey, which appears in their September/October issue</a>.</li> 
        <li>In April 2013, <a href="http://www.leidenranking.com/ranking" onclick="javascript:_gaq.push(['_trackEvent','outbound-article','http://www.leidenranking.com/ranking']);">UCR was ranked 35th overall in the Leiden Rankings, an annual survey by the Centre for Science and Technology Studies at Leiden University</a> that measures per capita scientific impact.</li> 
        <li>In May 2013, t<a href="http://ucrtoday.ucr.edu/15025">he New America Foundation named UCR a “Next Generation University” </a>for success in enhancing student outcomes and maintaining robust research profiles despite economic pressures.</li> 
        <li>On August 16, 2013, UCR was ranked among the top 150 universities in the world by the <a href="http://ucrtoday.ucr.edu/17049">Shanghai Jiao Tong University Academic Ranking of World Universities.</a></li> 
       </ul>

  var got_dates = false;
  var activedates = Array();

  var initsel = "00";

  function getCookie(cname) {
    var cval = $.cookie(cname);
    if (cval != null && cval != "") {
      return unescape($.cookie(cname));
    } else {
      return cval;
    }
  }

  function setCookie(cname, value, expiredays) {
    $.cookie(cname, escape(value), { expires: expiredays });
  }

  function set_tips (data) {
    $("#eventlist").html(data.eventlist);
    $("#eventtips").html(data.tiplist);
    $("#eventlist a").each(function() {
      var tip  = $(this).attr('rel');
      if (tip != "" && $(tip).length > 0) {
        var data  = $(tip).html();
		var evt   = tip.replace("#eventtip_","");
		var title = '<a class="evtdetlnk" href="/calendar/event_detail.html?eventid=' + evt + '" >Event Details</a>';
        $(this).qtip({
          content: {
            prerender: true,
            text: data,
            title: {
              text: title,
              button: '<img src="/images/btn_close.gif" alt="Close" />'
            }
          },
          position: {
		    adjust: { screen: true },
            corner: { target: 'bottomLeft', tooltip: 'topLeft' }
          },
          show: { delay: 500, solo: true },
          hide: false,
          style: 'caltip'
        });
	  }
    });
  }


  function update_calendar(){
    var vv = $(".selectboxoptions_check").children(".selected").children(".elmValue").textChildren({stringDelimiter: '|'});
	$('.calcodeselected').removeClass('calcodeselected');
    var pe = "";
	if (vv != "00") pe = "&pe=" + vv;

    var cdsel = $(".selectboxoptions_check").children(".selected").children(".elmValue").textChildren({outputType: 'array'});
    $.each(cdsel, function() {
      $('.' + this + '_codeactive').addClass('calcodeselected');
      //$('.' + this + '_codeactive').children().addClass('calcodeselected');
    });

    var ee = "";
    if (edd != "") ee = "&endd=" + edd;
    // alert(pe);
    $.getJSON(calserver + "ae_get_website_events.php?jsoncallback=?&startd="+ std + ee + pe, function(data) {
      set_tips(data);
    });
  }


  function resetCalendar(yr, mth) {
    $("#cal_left_div").hide();
    initsel = $("#eventtype").val();
    $("#eventtype").val("");
    if (initsel == "") {
      var cval = getCookie(cname);
      //alert("cookie: " + cval);
      if (cval != null && cval != "") {
        initsel = cval;
      } else {
        initsel = "00";
      }
    } else {
      setCookie(cname,initsel,expire);
    }
    //alert (initsel);
    var sel = "";
    if (initsel == "00") sel = "selected";
    $('#type_select_div').html('<select id = "type_select" name="type_select"  multiple="multiple">' +
                               '  <option>Select Event Types</option>' +
                               '  <option value="00" ' + sel + ' ><< All Events >></option>' +
                               '</select>');

    $('#day_select_div').html('<select id = "day_select" name="day_select" >' +
                              '  <option>Select Days</option>' +
                              '  <option value="00" ><< All Days >></option>' +
                              '  <option value="01" selected="selected">test day</option>' +
                              '</select>');

    selyr  = yr;
    selmth = mth;
    std = selyr + '-' + selmth + '-1';
    var endm = selmth + 3;
    var endy = selyr;
    if ( endm > 12 ) {
      endm = endm - 12;
      endy = endy + 1;
    }
    edd = endy + '-' + endm + '-1';
    //$("#mthsel").html('Selected ' + std + ', ' + edd);
    var pe = "";
    if (initsel != "00") pe = "&pe=" + initsel;
    $.getJSON(calserver + "ae_get_website_events.php?jsoncallback=?&startd="+ std + '&endd=' + edd + pe,
      function(data) {
        set_tips(data);
        var found = false;
        $.each(data.typelist, function(i,item){
          sel = "";
          if (initsel.indexOf(item.typ) >= 0) {
            sel = "selected";
            found = true;
          } else if ( initsel != "00" && initsel.length == 2 && initsel.substr(1,1) == "0" &&
                      initsel.substr(0,1) == item.typ.substr(0,1) ) {
            sel = "selected";
            found = true;
          }
          // alert(item.typ + " " + item.lbl);
          $("#type_select").append("<option value='" + item.typ + "' " + sel + " >" + item.lbl + "</option>");
        });
        // no matching event type, reset for next update
        if (initsel != "00" && !found) initsel = "00";

        $("#type_select").custSelectBox();
        $("#day_select").custSelectBox();
        $(".select_wrap > div").css('width','140px');
        $(".selectboxfoot > div").css('width','117px');
        $(".selectboxoptions_check").css('width','175px');
        $(".selectboxoptions_radio").css('width','175px');

        var selall = $(".selectboxoptions_check li:first");
        $(".selectboxoptions_check li:first").click( function() {
          if ($(this).hasClass("selected")) {
            $(".selectboxoptions_check li").removeClass("selected");
            $(".selectboxoptions_check li:first").addClass("selected");
          }
        });

        $("#type_select_div").mouseenter( function() {
          $(".selectboxoptions_check li").removeClass("hideitem");
        });

        $("#type_select_div").mouseleave( function() {
          $(".selectboxoptions_check li").addClass("hideitem");
        });

        $(".selectboxoptions_check").click( function() {
          var vv = $(this).children(".selected").children(".elmValue").textChildren({stringDelimiter: '|'});
          if (vv == "") {
            $(".selectboxoptions_check li:first").addClass("selected");
            vv = $(this).children(".selected").children(".elmValue").textChildren({stringDelimiter: '|'});
          }

          if (vv != "00") {
            $(".selectboxoptions_check li:first").removeClass("selected");
            vv = $(this).children(".selected").children(".elmValue").textChildren({stringDelimiter: '|'});
          }
          //alert(vv);
          setCookie(cname, vv, expire);
          update_calendar();
        });

        $(".selectboxoptions_radio li:first").click( function() {
          std = selyr + '-' + selmth + '-1';
          var endm = selmth + 3;
          var endy = selyr;
          if (endm > 12) {
            endm = endm - 12;
            endy = endy + 1;
          }
          edd = endy + '-' + endm + '-1';
          $("#day_select_div").hide();
          $("#type_select_div").show();
          update_calendar();
        });

        $("#day_select_div").mouseenter( function() {
          $(".selectboxoptions_radio li").removeClass("hideitem");
        });

        $("#day_select_div").mouseleave( function() {
          $(".selectboxoptions_radio li").addClass("hideitem");
        });

        $("#day_select_div").hide();
        $("#type_select_div").show();
        $("#cal_left_div").show();
        if (initsel != "00") {
          $('.calcodeselected').removeClass('calcodeselected');
          var cdsel = $(".selectboxoptions_check").children(".selected").children(".elmValue").textChildren({outputType: 'array'});
          $.each(cdsel, function() {
            $('.' + this + '_codeactive').addClass('calcodeselected');

          });
          //$('.' + initsel + '_codeactive').addClass('calcodeselected');
          //$('.' + initsel + '_codeactive').children().addClass('calcodeselected');
        }
    });
  }


    function init_calendar() {
      var i = 0;
      var zz = new Date();
      var z = activedates.length;
/*
      $("#datedata").html($("#datedata").html() + '<br>Len: ' + z  );
      for (i=0;i<z;i++) {
        zz = activedates[i];
        $("#datedata").html($("#datedata").html() + '<br>' + i  + ' ' + zz + ' ' + zz.getDate());
      }
*/
      $('#datep').datepicker({
        numberOfMonths: 3,
        maxDate: maxd,
        minDate: mind,
        dateFormat: 'D, M d, yy',
        altFormat: 'yy-mm-dd',
        altField: '#altdate',
        beforeShowDay: function(tdate) {
          var dd = tdate.getDate();
          var dm = tdate.getMonth();
          var dy = tdate.getFullYear();
          var dt = new Date();
          var cd = "";
          //$("#datedata").html($("#datedata").html() + dd + '-' + dm + '-' + dy + '   ');
          //$("#datedata").html($("#datedata").html() + '<br>' + tdate + ' ' + dd);

          for (i=0;i<z;i++) {
            dt = activedates[i];
            cd = activecodes[i];
            //$("#datedata").html($("#datedata").html() + i + ' ' + activedates[i] + ' ');
            if ( dd == dt.getDate() && dm == dt.getMonth() && dy == dt.getFullYear()) {
              //$("#datedata").html($("#datedata").html() + ' Match' );
	          return [true,cd];
	        }
          }
          return [false,''];
        },
        onSelect: function(dateText, inst) {
          $("#seldate").html(dateText);
          std = $("#altdate").val();
          edd = "";
          $(".selectboxoptions_check li").removeClass("selected");
          $(".selectboxoptions_check li:first").addClass("selected");
          $(".selectboxoptions_radio li").removeClass("selected");
          $(".selectboxoptions_radio li:last").addClass("selected");
          $(".selectboxoptions_radio li:last").html('<span class="elmValue">01</span>' + dateText);
          // $('.selectboxoptions_check li').attr('disabled',true);
          // $(".selectboxoptions_check li").addClass("disabled");
          $("#type_select_div").hide();
          $("#day_select_div").show();
          update_calendar();
        },
        onChangeMonthYear: function(yr, mth, inst) {
          resetCalendar(yr, mth);
        }
      });

      resetCalendar(mind.getFullYear(),mind.getMonth()+1);

      if (initsel != "00") {
	    $('.' + initsel + '_codeactive').addClass('calcodeselected');
        //$('.' + initsel + '_codeactive').children().addClass('calcodeselected');
	  }
    }

    function check_dates(){
      $('#datep').delay(100, function(){
        if (got_dates) {
          init_calendar();
        } else {
          check_dates();
        }
      });
    }

    $(document).ready(function(){
      $("#eventtips").hide();
      $("#cal_left_div").hide();
      calserver = $("#calserver").val();

      $.getJSON(calserver + "ae_get_event_dates.php?jsoncallback=?", function(data) {
        //alert("Got data" + data);
        var cdata = data.webcodes;
        var idx = 0;
        var idx2 = 0;
        var codes = "";

        $.each(data.webdates, function(i,item){
          //alert("dy " + item.dy );
          //var dt = new Date(item.mth + ' ' + item.dy + ', ' + item.yr);
          var dt = new Date(item.yr,item.mth-1,item.dy);
          idx = activedates.push(dt);
          codes = "";
          $.each(cdata, function(i,cd) {
            if (item.event_date == cd.event_date)  codes = codes + cd.prog_code + cd.event_code + "_codeactive ";
          });
          idx2 = activecodes.push(codes);
        });
        got_dates = true;
      });
      check_dates();
    });
